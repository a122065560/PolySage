### 3.2 PlatformAdapter 接口设计（MiniMax 方案）

**文件路径：** `platform_adapter.py`

```python
from abc import ABC, abstractmethod
from pathlib import Path
from typing import Optional, List
from dataclasses import dataclass, field

# ============ 数据结构 ============

@dataclass
class ChromeLaunchSpec:
    """Chrome 启动参数规格（跨平台统一）"""
    debug_port: int = 9222
    user_data_dir: Optional[Path] = None   # None → 用 default_user_data_dir()
    headless: bool = False
    extra_args: List[str] = field(default_factory=list)
    instance_id: str = "default"

@dataclass
class ChromeHandle:
    """Chrome 实例句柄，后续所有操作都靠它"""
    pid: int
    debug_port: int
    user_data_dir: Path
    cdp_url: str                            # ws://127.0.0.1:9222/...
    launch_time: float

# ============ 抽象接口 ============

class PlatformAdapter(ABC):
    """
    跨平台系统调用抽象层。
    所有 '操作系统/进程/文件系统' 相关代码必须走这里。
    业务代码禁止直接调 os.* / subprocess.* / pathlib 原生方法。
    """

    # ---- 路径与用户数据目录 ----

    @abstractmethod
    def app_data_dir(self) -> Path:
        """应用数据目录的跨平台等价物
        Mac:   ~/Library/Application Support/RoundTableAI
        Win:   %LOCALAPPDATA%/RoundTableAI
        Linux: ~/.local/share/RoundTableAI
        """

    @abstractmethod
    def default_user_data_dir(self) -> Path:
        """Chrome 用户数据目录（存 cookies/localStorage）
        Win 上必须放在 LOCALAPPDATA 下，不能在 ~ 下
        （避免 Windows Defender 实时扫描导致启动失败）
        """

    @abstractmethod
    def config_path(self) -> Path:
        """配置文件路径"""

    # ---- Chrome 生命周期 ----

    @abstractmethod
    def launch_chrome(self, spec: ChromeLaunchSpec) -> ChromeHandle:
        """启动 Chrome 并返回句柄
        Mac:  LaunchAgent + launchctl bootstrap
        Win:  subprocess.Popen + DETACHED_PROCESS + CREATE_NEW_PROCESS_GROUP
              必须强制加 --no-sandbox（Windows 容器/CI 环境 Chrome sandbox 默认失败）
        """

    @abstractmethod
    def kill_chrome_tree(self, handle: ChromeHandle, grace_seconds: float = 5.0) -> None:
        """递归杀 Chrome 进程树
        Mac:  kill -TERM → 5s 不响应 → kill -9
        Win:  首选 CDP 优雅关闭 → 5s 超时 → taskkill /F /T /PID
              （关键：/T 必须带，否则 GPU/Utility 子进程变孤儿）
        """

    @abstractmethod
    def is_port_listening(self, port: int) -> bool:
        """检查端口是否被占用"""

    @abstractmethod
    def is_chrome_alive(self, handle: ChromeHandle) -> bool:
        """检查 Chrome 是否存活"""

    # ---- 用户数据目录卫生 ----

    @abstractmethod
    def pre_launch_cleanup(self, user_data_dir: Path) -> None:
        """启动前清理上次异常退出的残留锁文件
        跨平台统一 glob: SingletonLock / SingletonCookie / SingletonSocket / LOCK
        Win 上额外 WAIT 500ms 再删，避免与 Chrome 自身释放竞态
        """

    @abstractmethod
    def post_close_cleanup(self, user_data_dir: Path) -> None:
        """关闭后清理（Chrome 偶尔留 .temp、Crashpad 残留）
        不删 cookies/history
        """

    @abstractmethod
    def claim_user_data_dir(self, user_data_dir: Path) -> None:
        """为这个 Chrome 实例打上 '我在用' 标记（写文件锁）
        防止主程序多实例误启动
        """

    @abstractmethod
    def release_user_data_dir(self, user_data_dir: Path) -> None:
        """释放 '我在用' 标记"""

    # ---- 进程级工具 ----

    @abstractmethod
    def find_processes_by_name(self, name_substr: str) -> List[int]:
        """按名称查找进程 PID
        Mac: pgrep -f; Win: psutil.process_iter
        """

    @abstractmethod
    def kill_process_tree_by_name(self, name_substr: str) -> int:
        """兜底：按名字暴力清除所有匹配进程，返回杀掉的数量
        用于 Chrome 异常但句柄丢失时的最后挣扎
        """

    # ---- 平台诊断 ----

    @abstractmethod
    def diagnose(self) -> dict:
        """健康检查，供 UI 设置页显示
        包含：OS / 版本 / 架构 / SIP 状态(Mac) / Defender 状态(Win) /
              Chrome 路径 / Playwright 浏览器路径 / 磁盘剩余 / 端口占用
        """

    @abstractmethod
    def name(self) -> str:
        """返回平台名称："macOS" / "Windows" / "Linux" """


# ============ 工厂自动选择 ============

_current: Optional[PlatformAdapter] = None

def current_adapter() -> PlatformAdapter:
    global _current
    if _current is None:
        import sys
        if sys.platform == "darwin":
            from .macos_adapter import MacOSAdapter as _Impl
        elif sys.platform == "win32":
            from .windows_adapter import WindowsAdapter as _Impl
        elif sys.platform.startswith("linux"):
            from .linux_adapter import LinuxAdapter as _Impl
        else:
            raise RuntimeError(f"Unsupported platform: {sys.platform}")
        _current = _Impl()
    return _current
```

### 3.3 PlatformAdapter —— Windows 端关键实现（Kimi 补充）

```python
# windows_adapter.py —— 关键实现片段

import os
import time
import subprocess
import psutil
from pathlib import Path

class WindowsAdapter(PlatformAdapter):
    def get_user_data_dir(self) -> Path:
        # 必须放在 LOCALAPPDATA，不能放 ~ 下（即 C:\Users\用户名）
        # 原因：Windows Defender 实时保护会高频扫描用户目录下的新建文件夹
        return Path(os.environ['LOCALAPPDATA']) / 'RoundTableAI' / 'chrome-data'

    def launch_chrome(self, spec: ChromeLaunchSpec) -> ChromeHandle:
        chrome_path = self._find_chrome_exe()
        user_data = spec.user_data_dir or self.get_user_data_dir()

        # Windows 关键 flags
        args = [
            str(chrome_path),
            f'--remote-debugging-port={spec.debug_port}',
            f'--user-data-dir={user_data}',
            '--no-first-run',
            '--no-default-browser-check',
            '--no-sandbox',          # Windows 容器/CI 环境必需
            '--disable-dev-shm-usage',
            '--disable-gpu',          # Windows 上 GPU 进程有时导致崩溃
        ]

        # DETACHED_PROCESS + CREATE_NEW_PROCESS_GROUP
        # 防止父进程退出时连带杀死 Chrome
        import ctypes
        CREATE_NEW_PROCESS_GROUP = 0x00000200
        DETACHED_PROCESS = 0x00000008

        proc = subprocess.Popen(
            args,
            creationflags=CREATE_NEW_PROCESS_GROUP | DETACHED_PROCESS,
            stdout=subprocess.DEVNULL,
            stderr=subprocess.DEVNULL,
        )

        return ChromeHandle(
            pid=proc.pid,
            debug_port=spec.debug_port,
            user_data_dir=user_data,
            cdp_url=f'ws://127.0.0.1:{spec.debug_port}/devtools/browser/',
            launch_time=time.time(),
        )

    def kill_chrome_tree(self, handle: ChromeHandle, grace_seconds: float = 5.0) -> None:
        # 三步清理：CDP 优雅关闭 → 终止进程树 → 清理锁文件

        # Step 1: 尝试 CDP 发送 Browser.close
        try:
            import requests
            requests.get(f'http://127.0.0.1:{handle.debug_port}/json/version', timeout=2)
            requests.post(
                f'http://127.0.0.1:{handle.debug_port}/devtools/browser/',
                json={'method': 'Browser.close'},
                timeout=5
            )
            time.sleep(2)
        except Exception:
            pass

        # Step 2: psutil 递归杀进程树（比 taskkill /F /T 更可控）
        try:
            parent = psutil.Process(handle.pid)
            children = parent.children(recursive=True)
            for child in children:
                child.terminate()
            parent.terminate()
            gone, alive = psutil.wait_procs(children + [parent], timeout=3)
            for p in alive:
                p.kill()
        except psutil.NoSuchProcess:
            pass

        # Step 3: 清理锁文件（Windows 上 SingletonLock 是文件，不是 socket）
        lock_files = ['SingletonLock', 'SingletonCookie', 'SingletonSocket', 'LOCK']
        for lock in lock_files:
            lock_path = handle.user_data_dir / lock
            if lock_path.exists():
                try:
                    lock_path.unlink()
                except PermissionError:
                    time.sleep(0.5)
                    lock_path.unlink()
```

## 四、构建与打包策略

### 4.1 整体设计（Kimi 方案）

- **零交叉编译**：macOS runner 只产 .dmg，Windows runner 只产 .exe
- **缓存加速**：pip + Playwright 浏览器双缓存，避免每次 3-5 分钟下载
- **产物自动归集**：构建完成后自动上传到 GitHub Releases

### 4.2 GitHub Actions Workflow 完整 YAML

**文件路径：** `.github/workflows/release.yml`

```yaml
name: Cross-Platform Release

on:
  workflow_dispatch:
    inputs:
      version:
        description: 'Release version (e.g., v1.2.0)'
        required: true
  push:
    tags:
      - 'v*'

jobs:
  build-macos:
    runs-on: macos-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - uses: actions/cache@v4
        with:
          path: |
            ~/Library/Caches/pip
            ~/Library/Caches/ms-playwright
          key: ${{ runner.os }}-deps-${{ hashFiles('requirements.txt') }}

      - run: pip install -r requirements.txt pyinstaller==6.3.0

      - name: Install Playwright browsers (cached)
        run: playwright install chrome

      - name: Build macOS .app
        run: |
          pyinstaller \
            --target-arch arm64 \
            --windowed \
            --osx-bundle-identifier com.roundtableai.app \
            --icon assets/AppIcon.icns \
            --name RoundTableAI \
            RoundTableAI.spec

      - name: Create DMG
        run: |
          brew install create-dmg
          create-dmg \
            --volname "RoundTableAI" \
            --window-pos 200 120 \
            --window-size 800 400 \
            --icon-size 100 \
            --app-drop-link 600 185 \
            "RoundTableAI-${{ github.event.inputs.version || github.ref_name }}-macOS.dmg" \
            "dist/RoundTableAI.app"

      - name: Codesign macOS (Ad-hoc for now)
        run: codesign --force --sign - "dist/RoundTableAI.app"

      - uses: actions/upload-artifact@v4
        with:
          name: macos-dmg
          path: "*.dmg"

  build-windows:
    runs-on: windows-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - uses: actions/cache@v4
        with:
          path: |
            ~\AppData\Local\pip\Cache
            ~\AppData\Local\ms-playwright
          key: ${{ runner.os }}-deps-${{ hashFiles('requirements.txt') }}

      - run: pip install -r requirements.txt pyinstaller==6.3.0

      - name: Install Playwright browsers
        run: playwright install chrome

      - name: Build Windows EXE
        run: |
          pyinstaller --windowed --name RoundTableAI RoundTableAI.spec

      - name: Build Windows installer with Inno Setup
        run: |
          # 使用 Inno Setup 命令行编译器
          iscc /DMyAppVersion="${{ github.event.inputs.version || github.ref_name }}" installer.iss

      - name: Code sign Windows binary
        shell: pwsh
        run: |
          $certBytes = [Convert]::FromBase64String("${{ secrets.WINDOWS_CERT_BASE64 }}")
          [IO.File]::WriteAllBytes("cert.pfx", $certBytes)
          signtool sign /f cert.pfx /p "${{ secrets.WINDOWS_CERT_PASSWORD }}" /tr http://timestamp.digicert.com /td sha256 /fd sha256 "dist/RoundTableAI.exe"
          signtool sign /f cert.pfx /p "${{ secrets.WINDOWS_CERT_PASSWORD }}" /tr http://timestamp.digicert.com /td sha256 /fd sha256 "*.exe"

      - uses: actions/upload-artifact@v4
        with:
          name: windows-exe
          path: |
            dist/*.exe
            *.exe

  release:
    needs: [build-macos, build-windows]
    runs-on: ubuntu-latest
    steps:
      - uses: actions/download-artifact@v4

      - uses: softprops/action-gh-release@v2
        with:
          files: |
            macos-dmg/*.dmg
            windows-exe/*.exe
          tag_name: ${{ github.event.inputs.version || github.ref_name }}
          generate_release_notes: true
```

VERSION=${1:-""}
if [ -z "$VERSION" ]; then
    echo "Usage: ./publish.sh v1.2.0"
    exit 1

# 1. 版本 bump
sed -i '' "s/version=.*/version=\"$VERSION\"/" src/__version__.py
git add -A

git push origin "$VERSION"

echo "✅ Tag $VERSION pushed. GitHub Actions building..."

# 2. 等待并下载产物（需安装 GitHub CLI: brew install gh）
gh run watch --repo=$(gh repo view --json nameWithOwner -q .nameWithOwner) --exit-status
gh run download --repo=$(gh repo view --json nameWithOwner -q .nameWithOwner) \
    --name macos-dmg \
    --name windows-exe \
    --dir ./dist/

echo "✅ 双平台产物已下载至 ./dist/"
echo "   macOS: ./dist/RoundTableAI-${VERSION}-macOS.dmg"
echo "   Windows: ./dist/RoundTableAI-${VERSION}-Windows.exe"

## 六、平台差异处理对照表

| 功能点 | macOS 实现 | Windows 实现 | 注意事项 |
|--------|-----------|--------------|----------|
| Chrome 启动 | LaunchAgent + launchctl bootstrap | subprocess.Popen + DETACHED_PROCESS + CREATE_NEW_PROCESS_GROUP | Win 必须带 `--no-sandbox` |
| Chrome 关闭 | `kill -TERM` → 超时 → `kill -9` | CDP 优雅关闭 → 超时 → `taskkill /F /T /PID` | Win 的 `/T` 不能漏 |
| 用户数据目录 | `~/Library/Application Support/RoundTableAI` | `%LOCALAPPDATA%/RoundTableAI` | Win 不能在 `~` 下，触发 Defender 扫描 |
| 残留锁清理 | glob 删除 Singleton* 文件 | glob 删除 Singleton* + 500ms 延迟 | Win 需延迟避免竞态 |
| 路径分隔符 | `/` | `\` | 全部使用 `pathlib.Path` |
| 代码签名 | `codesign` | `signtool`（需要 EV 证书） | Win 必须 EV 证书 |

## 九、后续开发流程

### 日常开发（macOS）
```bash
# 1. 写代码 / AI 生成代码
# 2. 本地测试
python main.py

# 3. 提交
git add -A
git commit -m "feat: xxx"
git push
```

### 发布流程
```bash
# 本地一条命令，触发双平台构建
./publish.sh v1.2.0

# 产物自动下载到 ./dist/
# macOS: ./dist/RoundTableAI-1.2.0-macOS.dmg
# Windows: ./dist/RoundTableAI-1.2.0-Windows.exe
```

### 添加新 AI 平台
1. 在 `config.json` 的 `ai_platforms` 中新增一条 JSON 配置
2. 如需要特殊选择器，在 `browser.py` 中新增一个 if 分支（AI 可生成）
3. 本地测试 → `git push` → CI 自动产出双平台新版本
